-- setup vim plug
local Plug = vim.fn["plug#"]
vim.call('plug#begin','~/.config/nvim/plugged')
  Plug 'jalvesaq/Nvim-R' -- R support
  Plug 'dense-analysis/ale' -- autoconfigured for lintr
  Plug 'junegunn/fzf.vim' -- fuzzy finding
vim.call('plug#end')
